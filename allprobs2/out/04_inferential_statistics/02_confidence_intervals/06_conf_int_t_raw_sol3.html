    x = c(282, 312, 279, 315, 301, 307, 340, 323, 277, 212, 246, 334, 356, 329)
n = length(x)
xbar = mean(x)
s = sd(x)
SE = s/sqrt(n)
gamma = 0.8   # Probability T is between -tstar and tstar
cumulative = gamma+(1-gamma)/2   # Probability T is less than tstar
tstar = qt(cumulative, n-1)
ME = tstar*s/sqrt(n)
LB = xbar-ME
UB = xbar+ME
print(data.frame(n,xbar,s,SE,tstar,ME,LB,UB,row.names=""))
