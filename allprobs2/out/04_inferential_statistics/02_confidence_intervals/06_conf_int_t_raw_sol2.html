    x = c(276, 369, 346, 316, 285, 278, 349, 283)
n = length(x)
xbar = mean(x)
s = sd(x)
SE = s/sqrt(n)
gamma = 0.9   # Probability T is between -tstar and tstar
cumulative = gamma+(1-gamma)/2   # Probability T is less than tstar
tstar = qt(cumulative, n-1)
ME = tstar*s/sqrt(n)
LB = xbar-ME
UB = xbar+ME
print(data.frame(n,xbar,s,SE,tstar,ME,LB,UB,row.names=""))
